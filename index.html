<script>

    function calcularTaxas() {

        const banco = document.getElementById('bank').value;

        const mensalidade = parseFloat(document.getElementById('monthly_fee').value);

        const valorPIX = parseFloat(document.getElementById('pix_value').value);

        const valorTED = parseFloat(document.getElementById('ted_value').value);

        const qtdPIX = parseInt(document.getElementById('pix_count').value);

        const qtdTED = parseInt(document.getElementById('ted_count').value);


        if (isNaN(mensalidade) || isNaN(valorPIX) || isNaN(valorTED) || isNaN(qtdPIX) || isNaN(qtdTED)) {

            alert("Por favor, preencha todos os campos.");

            return;

        }


        let totalPIX = valorPIX * qtdPIX;

        let totalTED = valorTED * qtdTED;

        let custoTotal = mensalidade + totalPIX + totalTED;


        let taxaBanco = 0;


        // Aqui você pode definir as taxas específicas para cada banco

        if (banco === "banco1") {

            taxaBanco = 10; // Exemplo de taxa do Banco 1

        } else if (banco === "banco2") {

            taxaBanco = 15; // Exemplo de taxa do Banco 2

        } else if (banco === "banco3") {

            taxaBanco = 20; // Exemplo de taxa do Banco 3

        }


        let custoComTaxa = custoTotal + taxaBanco;


        document.getElementById('result').innerHTML = `

            <h3>Resultado:</h3>

            <p><strong>Banco:</strong> ${banco.charAt(0).toUpperCase() + banco.slice(1)}</p>

            <p><strong>Mensalidade:</strong> R$ ${mensalidade.toFixed(2)}</p>

            <p><strong>Total de PIX (R$):</strong> R$ ${totalPIX.toFixed(2)}</p>

            <p><strong>Total de TED (R$):</strong> R$ ${totalTED.toFixed(2)}</p>

            <p><strong>Custo Total (R$):</strong> R$ ${custoTotal.toFixed(2)}</p>

            <p><strong>Taxa do Banco (R$):</strong> R$ ${taxaBanco.toFixed(2)}</p>

            <p><strong>Custo Total com Taxa (R$):</strong> R$ ${custoComTaxa.toFixed(2)}</p>

        `;

    }

</script>
